### 跨框架通信

```typescript
// Vue组件发布事件
import { EventBus } from '@/shared/eventBus';
EventBus.emit('strategy:selected', { name: 'MACD', params: {...} });

// React组件订阅事件
EventBus.on('strategy:selected', (data) => {
  console.log('Selected strategy:', data.name);
});

// 共享状态
sharedStore.set('selectedSymbol', 'BTC/USDT');
sharedStore.subscribe('selectedSymbol', (state) => {
  console.log('Symbol changed:', state.selectedSymbol);
});
```
